Lab-14: MongoDB Aggregate Method (MongoDB)



1. Fetch each student with matching department details.

db.Student.aggregate([
{
$lookup:{
from:"Department",
localField:"dept_id",
foreignField:"_id",
as:"department"
}
}
])



2. Display student name and department name.

db.Student.aggregate([
{
$lookup:{
from:"Department",
localField:"dept_id",
foreignField:"_id",
as:"department"
}
},
{$unwind:"$department"},
{
$project:{
_id:0,
name:1,
department:"$department.dept_name"
}
}
])



3. Retrieve every student with fee information.

db.Student.aggregate([
{
$lookup:{
from:"Fees",
localField:"_id",
foreignField:"student_id",
as:"fees"
}
}
])



4. Find students whose fee status is Unpaid.

db.Student.aggregate([
{
$lookup:{
from:"Fees",
localField:"_id",
foreignField:"student_id",
as:"fees"
}
},
{$unwind:"$fees"},
{$match:{"fees.status":"Unpaid"}}
])



5. Fetch students with department and fee records in one pipeline.

db.Student.aggregate([
{
$lookup:{
from:"Department",
localField:"dept_id",
foreignField:"_id",
as:"department"
}
},
{
$lookup:{
from:"Fees",
localField:"_id",
foreignField:"student_id",
as:"fees"
}
}
])



6. Return students with no matching department.

db.Student.aggregate([
{
$lookup:{
from:"Department",
localField:"dept_id",
foreignField:"_id",
as:"department"
}
},
{$match:{department:{$eq:[]}}}
])



7. Select students whose department is Computer Science.

db.Student.aggregate([
{
$lookup:{
from:"Department",
localField:"dept_id",
foreignField:"_id",
as:"department"
}
},
{$unwind:"$department"},
{$match:{"department.dept_name":"Computer Science"}}
])



8. Join departments and sort students by marks (descending).

db.Student.aggregate([
{
$lookup:{
from:"Department",
localField:"dept_id",
foreignField:"_id",
as:"department"
}
},
{$sort:{marks:-1}}
])



9. Count students in each department.

db.Student.aggregate([
{
$lookup:{
from:"Department",
localField:"dept_id",
foreignField:"_id",
as:"department"
}
},
{$unwind:"$department"},
{
$group:{
_id:"$department.dept_name",
total_students:{$sum:1}
}
}
])



10. Students with marks greater than 80.

db.Student.aggregate([
{
$lookup:{
from:"Department",
localField:"dept_id",
foreignField:"_id",
as:"department"
}
},
{$match:{marks:{$gt:80}}}
])



11. Maximum marks in each department.

db.Student.aggregate([
{
$lookup:{
from:"Department",
localField:"dept_id",
foreignField:"_id",
as:"department"
}
},
{$unwind:"$department"},
{
$group:{
_id:"$department.dept_name",
max_marks:{$max:"$marks"}
}
}
])



12. Students whose fee status is Paid.

db.Student.aggregate([
{
$lookup:{
from:"Fees",
localField:"_id",
foreignField:"student_id",
as:"fees"
}
},
{$unwind:"$fees"},
{$match:{"fees.status":"Paid"}}
])



13. Calculate total fee amount per student.

db.Student.aggregate([
{
$lookup:{
from:"Fees",
localField:"_id",
foreignField:"student_id",
as:"fees"
}
},
{$unwind:"$fees"},
{
$group:{
_id:"$name",
total_fee:{$sum:"$fees.amount"}
}
}
])



14. Fetch each department with all students in that department.

db.Department.aggregate([
{
$lookup:{
from:"Student",
localField:"_id",
foreignField:"dept_id",
as:"students"
}
}
])


15. Returns only those departments which do not have any students mapped to them.

db.Department.aggregate([
{
$lookup:{
from:"Student",
localField:"_id",
foreignField:"dept_id",
as:"students"
}
},
{
$match:{
students: { $eq: [] }
}
}
])
